$.toggleList()

Toggle list using [target, index, callback]:

  • 1
  • 2
  • 3
  • 1
  • 2
  • 3
    <script type="text/javascript">
    $(function() {
        $('.ul1 > li').toggleList({
            target: $('.ul2 > li'),
            callback: function(target, i) {
                target.css('background', ['#CEEDF1', '#3DDD80', '#ECFF6A'][i]);
            }
        });
    });
    </script>

    <ul class="ul1">
        <li>1</li>
        <li>2</li>
        <li>3</li>
    </ul>

    <ul class="ul2">
        <li class="active">1</li>
        <li>2</li>
        <li>3</li>
    </ul>

    <script type="text/javascript">
    $(function() {
        $('.ul2 > li').toggleList({
            target: $('.ul2 > li'),
            type: 'class',
            event: 'mouseover',
            callback: function(target, i) {
                target.css('background', ['#CEEDF1', '#3DDD80', '#ECFF6A'][i]);
            }
        });
    });
    </script>

    <ul class="ul1">
        <li class="foo active" >1</li>
        <li class="bar" >2</li>
        <li class="baz" >3</li>
    </ul>

    <ul class="ul2">
        <li class="foo active">1</li>
        <li class="bar" >2</li>
        <li class="baz" >3</li>
    </ul>
